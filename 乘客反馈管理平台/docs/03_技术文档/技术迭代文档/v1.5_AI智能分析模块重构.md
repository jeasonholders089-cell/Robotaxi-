## 1. 概述

### 1.1 技术背景

v1.5 版本对 AI 智能分析模块进行重构，实现完整的多步骤分析流水线，支持异步任务执行和实时进度反馈。

### 1.2 核心改动

| 改动点 | 当前实现 | 目标实现 |
|--------|----------|----------|
| 分析入口 | 自动加载数据 | 点击"开始分析"按钮触发 |
| 数据筛选 | 无直接关联 FilterBar | 完全基于 FilterBar 筛选条件 |
| 数据规模 | ≤100 条 | 默认 100 条，最多 2000 条 |
| 分析流程 | 单次调用 | 数据清洗 → 摘要 → 问题分析 → 建议 |
| 任务执行 | 同步 | 异步任务 + 进度轮询 |
| 结果展示 | 两个独立模块 | 三块结果：摘要、问题分类、优化建议 |

### 4.2 分析流水线

```python
async def run_analysis_task(task_id: str, filters: dict, db: AsyncSession):
    """
    分析流水线：

    Step 1 [0-10%]:  根据筛选条件查询反馈数据
    Step 2 [10-30%]: 数据清洗
    Step 3 [30-60%]: 生成 AI 摘要
    Step 4 [60-85%]: 产品问题分析
    Step 5 [85-100%]: 生成优化建议
    """

    # Step 1: 获取数据
    update_task_progress(task_id, 5)
    feedbacks, stats = await service._get_feedback_data_for_analysis(
        start_date=filters.get("start_date"),
        end_date=filters.get("end_date"),
        city=filters.get("city"),
        rating_min=filters.get("rating_min"),
        rating_max=filters.get("rating_max"),
        status=filters.get("status"),
        keyword=filters.get("keyword"),
        feedback_type=filters.get("feedback_type"),
        max_count=2000,
    )
    update_task_progress(task_id, 10)

    # Step 2: 数据清洗
    update_task_progress(task_id, 15)
    cleaned_data = clean_feedbacks(feedbacks)
    update_task_progress(task_id, 30)

    # Step 3: 生成摘要
    summary = await ai_client.summarize(cleaned_data, stats)
    update_task_progress(task_id, 60)

    # Step 4: 产品问题分析
    problems = await ai_client.analyze_problems(cleaned_data)
    update_task_progress(task_id, 85)

    # Step 5: 生成建议
    suggestions = await ai_client.generate_suggestions(problems, cleaned_data)
    update_task_progress(task_id, 100)

    # 保存结果
    set_task_result(task_id, summary, problems, suggestions, len(cleaned_data))
```

### 4.3 数据清洗逻辑

```python
def clean_feedbacks(feedbacks: List[Dict]) -> List[Dict]:
    """
    清洗反馈数据：

    1. 去除无效反馈（空文本、纯表情等）
    2. 按评分/时间排序
    3. 提取统计数据
    """
    cleaned = []
    for fb in feedbacks:
        text = fb.get("feedback_text", "").strip()
        # 跳过空文本或纯表情
        if not text or is_emoji_only(text):
            continue
        cleaned.append(fb)

    # 按评分升序、反馈时间倒序排列（低分优先）
    cleaned.sort(key=lambda x: (x["rating"], x["trip_time"]), reverse=[False, True])

    return cleaned
```

async function pollTaskStatus(taskId: string) {
  const interval = 3000; // 3秒轮询

  const timer = setInterval(async () => {
    const response = await fetch(`/api/ai/analyze/${taskId}`);
    const data = await response.json();

    if (data.status === 'completed' || data.status === 'failed') {
      clearInterval(timer);
      // 更新状态
    }
  }, interval);
}
