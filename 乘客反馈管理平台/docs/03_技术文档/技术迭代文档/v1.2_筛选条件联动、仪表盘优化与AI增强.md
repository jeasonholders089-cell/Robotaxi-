const routeDistribution = distributionData?.route_distribution?.slice(0, 10).map((item) => ({

### 3.2 时段热力图

**ECharts 配置**：
```typescript
{
  type: 'heatmap',
  data: hourDistribution.map((item) => [item.hour, 0, item.count]),
  xAxis: { type: 'category', data: [...Array(24).keys()] },
  yAxis: { type: 'category', data: [''] },
  visualMap: {
    min: 0,
    max: maxCount,
    calculable: true,
    orient: 'horizontal',
    left: 'center',
    inRange: { color: ['#F5F5F5', '#E5F7FF', '#91D5FF', '#1890FF', '#0050B3'] }
  }
}
```

### 3.3 筛选联动

// FeedbackManagement.tsx
const { data: stats } = useOverviewStats({
