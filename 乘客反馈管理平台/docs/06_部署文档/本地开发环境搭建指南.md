# 本地开发环境搭建指南

## 1. 使用 Docker 启动本地 MySQL

### 1.1 启动 MySQL 容器

```bash
# 拉取 MySQL 8.0 镜像
docker pull mysql:8.0

# 启动 MySQL 容器
docker run -d \
  --name robotaxi-mysql \
  -p 3306:3306 \
  -e MYSQL_ROOT_PASSWORD=root_password \
  -e MYSQL_DATABASE=robotaxi \
  -e MYSQL_USER=robotaxi \
  -e MYSQL_PASSWORD=dev_password \
  -v $(pwd)/mysql_data:/var/lib/mysql \
  mysql:8.0 \
  --character-set-server=utf8mb4 \
  --collation-server=utf8mb4_unicode_ci
```

### 1.2 验证 MySQL 连接

```bash
# 进入容器
docker exec -it robotaxi-mysql mysql -uroot -proot_password

# 或使用本地客户端
mysql -h 127.0.0.1 -uroot -proot_password
```

## 2. 导入数据库表结构

### 2.1 创建数据库和表

```bash
# 方式一：命令行执行 SQL
docker exec -i robotaxi-mysql mysql -uroot -proot_password < docs/03_技术文档/init.sql

# 方式二：进入容器后执行
docker exec -it robotaxi-mysql mysql -uroot -proot_password
```

### 2.2 init.sql 文件内容

```sql
-- 创建数据库
CREATE DATABASE IF NOT EXISTS robotaxi DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
USE robotaxi;

-- 反馈主表
CREATE TABLE `feedback` (
  `id` VARCHAR(20) NOT NULL COMMENT '反馈ID，主键，格式：FB+日期+序号',
  `trip_id` VARCHAR(20) NOT NULL COMMENT '行程ID',
  `passenger_id` VARCHAR(20) NOT NULL COMMENT '乘客ID（脱敏）',
  `vehicle_id` VARCHAR(30) NOT NULL COMMENT '车辆ID，格式：Robotaxi-{城市}-{序号}',
  `city` VARCHAR(20) NOT NULL COMMENT '城市',
  `route` VARCHAR(100) NOT NULL COMMENT '路线名称',
  `route_start` VARCHAR(50) NOT NULL COMMENT '起点',
  `route_end` VARCHAR(50) NOT NULL COMMENT '终点',
  `trip_time` DATETIME NOT NULL COMMENT '行程开始时间',
  `trip_duration` INT NOT NULL COMMENT '行程时长（分钟）',
  `rating` TINYINT NOT NULL COMMENT '评分 1-5',
  `feedback_text` TEXT NOT NULL COMMENT '乘客原始反馈',
  `feedback_type` JSON DEFAULT NULL COMMENT 'AI分类结果，数组格式',
  `sentiment` VARCHAR(10) DEFAULT NULL COMMENT '情感倾向：正面/中性/负面',
  `keywords` JSON DEFAULT NULL COMMENT 'AI提取的关键词',
  `status` VARCHAR(20) NOT NULL DEFAULT 'pending' COMMENT '跟进状态',
  `handler` VARCHAR(50) DEFAULT NULL COMMENT '处理人',
  `handler_notes` TEXT DEFAULT NULL COMMENT '处理记录',
  `handled_at` DATETIME DEFAULT NULL COMMENT '最近处理时间',
  `ai_summary` TEXT DEFAULT NULL COMMENT '该反馈的AI摘要',
  `feedback_channel` VARCHAR(16) DEFAULT 'App' COMMENT '反馈渠道：App/小程序/电话/Web',
  `reply_text` TEXT DEFAULT NULL COMMENT '客服回复内容',
  `reply_time` DATETIME DEFAULT NULL COMMENT '客服回复时间',
  `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '记录创建时间',
  `updated_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '最后修改时间',
  PRIMARY KEY (`id`),
  INDEX `idx_trip_time` (`trip_time`),
  INDEX `idx_rating` (`rating`),
  INDEX `idx_city` (`city`),
  INDEX `idx_status` (`status`),
  INDEX `idx_created_at` (`created_at`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='反馈主表';

-- 路线预设表
CREATE TABLE `route_preset` (
  `id` INT NOT NULL AUTO_INCREMENT COMMENT '主键',
  `city` VARCHAR(20) NOT NULL COMMENT '城市',
  `route_name` VARCHAR(100) NOT NULL COMMENT '路线名称',
  `route_start` VARCHAR(50) NOT NULL COMMENT '起点',
  `route_end` VARCHAR(50) NOT NULL COMMENT '终点',
  `active` TINYINT NOT NULL DEFAULT 1 COMMENT '是否启用',
  `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_route` (`city`, `route_start`, `route_end`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='路线预设表';
```

## 3. 导入模拟数据

### 3.1 转换 JSON 为 SQL INSERT

mock 数据是 JSON 格式，需要转换为 SQL：

```python
# scripts/convert_mock_data.py
import json

with open('data/mock/feedback_data.json', 'r', encoding='utf-8') as f:
    data = json.load(f)

sql_statements = []
for record in data['data']:
    sql = f"""INSERT INTO feedback (
        id, trip_id, passenger_id, vehicle_id, city, route,
        route_start, route_end, trip_time, trip_duration, rating,
        feedback_text, feedback_type, sentiment, keywords, status,
        handler, handler_notes, handled_at, ai_summary,
        feedback_channel, reply_text, reply_time, created_at, updated_at
    ) VALUES (
        '{record['feedback_no']}',
        '{record['trip_id']}',
        '{record['passenger_id']}',
        '{record['vehicle_id']}',
        '{record['city']}',
        '{record['route_start']} → {record['route_end']}',
        '{record['route_start']}',
        '{record['route_end']}',
        '{record['trip_time']}',
        {record['trip_duration']},
        {record['rating']},
        '{record['feedback_text'].replace("'", "\\'")}',
        '["' + '","'.join(record.get('ai_category', [])) + '"]',
        NULL,
        '["' + '","'.join(record.get('ai_keywords', [])) + '"]',
        '{record['status']}',
        NULL, NULL, NULL, NULL,
        '{record.get('feedback_channel', 'App')}',
        '{record.get('reply_text', '')}',
        '{record.get('reply_time', '')}',
        '{record['trip_time']}',
        '{record['trip_time']}'
    );"""
    sql_statements.append(sql)

with open('data/mock/feedback_data.sql', 'w', encoding='utf-8') as f:
    f.write('\n'.join(sql_statements))
```

### 3.2 导入 SQL 文件

```bash
# 导入模拟数据
docker exec -i robotaxi-mysql mysql -uroot -proot_password robotaxi < data/mock/feedback_data.sql

# 验证导入结果
docker exec -it robotaxi-mysql mysql -uroot -proot_password -e "SELECT COUNT(*) FROM robotaxi.feedback;"
```

## 5. 数据迁移到阿里云

### 5.1 导出本地数据

```bash
# 导出为 SQL 文件
docker exec robotaxi-mysql mysqldump -uroot -proot_password \
  --databases robotaxi \
  --single-transaction \
  --quick \
  --lock-tables=false \
  > backup_$(date +%Y%m%d).sql

# 压缩传输
gzip backup_20260416.sql
```

### 5.2 传输到阿里云

```bash
# 方式一：scp 传输
scp backup_20260416.sql.gz root@你的服务器IP:/tmp/

# 方式二：使用 rsync
rsync -avz backup_20260416.sql.gz root@你的服务器IP:/tmp/
```

### 5.3 导入到阿里云 MySQL

```bash
# 登录阿里云服务器
ssh root@你的服务器IP

# 解压
gunzip /tmp/backup_20260416.sql.gz

# 导入（先备份生产数据）
mysqldump -uroot -p生产密码 robotaxi > backup_production_$(date +%Y%m%d).sql
mysql -uroot -p生产密码 robotaxi < /tmp/backup_20260416.sql
```

## 6. Docker Compose 一键启动（推荐）

### 6.1 创建 docker-compose.yml

```yaml
# docker-compose.yml
version: '3.8'

services:
  mysql:
    image: mysql:8.0
    container_name: robotaxi-mysql
    ports:
      - "3306:3306"
    environment:
      MYSQL_ROOT_PASSWORD: root_password
      MYSQL_DATABASE: robotaxi
      MYSQL_USER: robotaxi
      MYSQL_PASSWORD: dev_password
    volumes:
      - ./mysql_data:/var/lib/mysql
      - ./init.sql:/docker-entrypoint-initdb.d/init.sql
    command: --character-set-server=utf8mb4 --collation-server=utf8mb4_unicode_ci

  backend:
    build:
      context: ./server
      dockerfile: Dockerfile.dev
    container_name: robotaxi-backend
    ports:
      - "8000:8000"
    environment:
      DATABASE_URL: mysql+pymysql://root:root_password@mysql:3306/robotaxi
      KIMI_API_KEY: ${KIMI_API_KEY}
      MINIMAX_API_KEY: ${MINIMAX_API_KEY}
    depends_on:
      - mysql
```

### 6.2 启动

```bash
# 启动所有服务
docker-compose up -d

# 查看日志
docker-compose logs -f

# 停止
docker-compose down
```

## 7. 注意事项

| 问题 | 解决方案 |
|------|----------|
| 本地端口 3306 被占用 | 修改 ports 映射为 `3307:3306` |
| Docker 数据持久化 | 确保 `mysql_data` 目录已挂载 |
| 中文乱码 | 确保 `--character-set-server=utf8mb4` |
| 导入慢 | 使用 `--single-transaction --quick` 参数 |
